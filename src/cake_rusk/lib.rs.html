#[proc_macro_attribute]
pub fn contract(attr: TokenStream, item: TokenStream) -> TokenStream {
    let result = contract::contract(attr, item);
    result
}

#[proc_macro_attribute]
pub fn cake(attr: TokenStream, item: TokenStream) -> TokenStream {
    let ident = syn::parse_macro_input!(attr as syn::Ident);
    let ident: &str = &ident.to_string();

    match ident {
        "storage" => {}
        _ => panic!("Unknown attribute \"{}\"", ident),
    }

    item
}
